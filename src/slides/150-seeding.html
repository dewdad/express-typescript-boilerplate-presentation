Seeding

1. Factories

A factory generates a new entity using fake data.

This can be used to seed data into the database or for testing.


  define(User, (faker: typeof Faker, settings: { roles: string[] }) => {
    const gender = faker.random.number(1);
    const firstName = faker.name.firstName(gender);
    const lastName = faker.name.lastName(gender);
    const email = faker.internet.email(firstName, lastName);

    const user = new User(); user.firstName = firstName; user.lastName = lastName;
    user.email = email; user.roles = settings.roles; return user;
  });
    

  define(Pet, (faker: typeof Faker, settings: any) => {
    const gender = faker.random.number(1);
    const name = faker.name.firstName(gender);

    const pet = new Pet();
    pet.name = name;
    pet.age = faker.random.number();
    pet.user = factory(User)({ roles: ['admin'] })
    return pet;
  });
